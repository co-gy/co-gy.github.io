for t in range(max_iteration):
    for s in env.state_space:
        for a in env.action_space:
            next_state, reward = env._get_next_state_and_reward(s, a)
            q_k[s][a] = reward + discount_rate * v_k[next_state]
        max_action_value = max(q_k[s], key=lambda _a: q_k[s][_a])
        policy[s] = max_action_value  # policy update
        v_k[s] = max(q_k[s].values())  # value update

for t in range(max_iteration):
    # policy evaluation
    for _ in range(100):
        for s in env.state_space:
            a = policy[s]
            next_state, reward = env._get_next_state_and_reward(s, a)
            v_pi_k[s] = reward + discount_rate*v_pi_k[next_state]
    # policy improvement
    for s in env.state_space:
        for a in env.action_space:
            next_state, reward = env._get_next_state_and_reward(s, a)
            q_k[s][a] = reward + discount_rate * v_pi_k[next_state]
        max_action_value = max(q_k[s], key=lambda _a: q_k[s][_a])
        policy[s] = max_action_value

运行结果

设置的max_iteration为20，能够训练出正确的policy，对比value iteration迭代次数为20，policy iteration收敛的更快，也算是符合预期

结尾
